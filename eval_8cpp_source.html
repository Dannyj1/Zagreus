26int Evaluation::evaluate() {
27 const int phase = calculatePhase();
28 const int modifier = board.getSideToMove() == WHITE ? 1 : -1;
29
30 evaluateMaterial();
31
32 const int whiteScore = ((whiteMidgameScore * (256 - phase)) + (whiteEndgameScore * phase)) / 256;
33 const int blackScore = ((blackMidgameScore * (256 - phase)) + (blackEndgameScore * phase)) / 256;
34
35 return (whiteScore - blackScore) * modifier;
36}

38constexpr int knightPhase = 1;
39constexpr int bishopPhase = 1;
40constexpr int rookPhase = 2;
41constexpr int queenPhase = 4;
42constexpr int totalPhase = knightPhase * 4 + bishopPhase * 4 + rookPhase * 4 + queenPhase * 2;
43
44int Evaluation::calculatePhase() const {
45 int phase = totalPhase;
46
47 phase -= (board.getPieceCount<WHITE_KNIGHT>() + board.getPieceCount<BLACK_KNIGHT>()) * knightPhase;
48 phase -= (board.getPieceCount<WHITE_BISHOP>() + board.getPieceCount<BLACK_BISHOP>()) * bishopPhase;
49 phase -= (board.getPieceCount<WHITE_ROOK>() + board.getPieceCount<BLACK_ROOK>()) * rookPhase;
50 phase -= (board.getPieceCount<WHITE_QUEEN>() + board.getPieceCount<BLACK_QUEEN>()) * queenPhase;
51
52 return (phase * 256 + (totalPhase / 2)) / totalPhase;
53}

55void Evaluation::evaluateMaterial() {

57 const Piece piece = static_cast<Piece>(pieceIndex);
