39uint64_t perft(Board& board, const int depth, bool printNodes) {
40 assert(depth >= 0);
41
42 if (depth == 0) {
43 return 1;
44 }
45
46 uint64_t nodes = 0;
47 MoveList moveList{};
48 const PieceColor sideToMove = board.getSideToMove();
49
50 if (sideToMove == WHITE) {
51 generateMoves<WHITE, ALL>(board, moveList);
52 } else {
53 generateMoves<BLACK, ALL>(board, moveList);
54 }
55
56 for (int i = 0; i < moveList.size; i++) {
57 board.makeMove(moveList.moves[i]);
58
59 if (sideToMove == WHITE) {
60 if (!board.isPositionLegal<WHITE>()) {
61 board.unmakeMove();
62 continue;
63 }
64 } else {
65 if (!board.isPositionLegal<BLACK>()) {
66 board.unmakeMove();
67 continue;
68 }
69 }
70
71 uint64_t perftNodes = perft(board, depth - 1, false);
72
73 if (printNodes) {
74 std::string moveNotation = getMoveNotation(moveList.moves[i]);
75
76 std::cout << moveNotation << ": " << perftNodes << std::endl;
77 }
78
79 nodes += perftNodes;
80 board.unmakeMove();
81 }
82
83 return nodes;
84}
