68int pop_1st_bit(uint64_t* bb) {
69 uint64_t b = *bb ^ (*bb - 1);
70 unsigned int fold = static_cast<unsigned>((b & 0xffffffff) ^ (b >> 32));
71 *bb &= (*bb - 1);
72 return BitTable[(fold * 0x783a9b23) >> 26];
73}

75uint64_t index_to_uint64_t(int index, int bits, uint64_t m) {
76 int i, j;
77 uint64_t result = 0ULL;
78 for (i = 0; i < bits; i++) {
79 j = pop_1st_bit(&m);
80 if (index & (1 << i)) result |= (1ULL << j);
81 }
82 return result;
83}

85uint64_t rmask(int sq) {
86 uint64_t result = 0ULL;
87 int rk = sq / 8, fl = sq % 8, r, f;
88 for (r = rk + 1; r <= 6; r++) result |= (1ULL << (fl + r * 8));
89 for (r = rk - 1; r >= 1; r--) result |= (1ULL << (fl + r * 8));
90 for (f = fl + 1; f <= 6; f++) result |= (1ULL << (f + rk * 8));
91 for (f = fl - 1; f >= 1; f--) result |= (1ULL << (f + rk * 8));
92 return result;
93}

149int transform(uint64_t b, uint64_t magic, int bits) {
150#if defined(USE_32_BIT_MULTIPLICATIONS)
151 return (unsigned)((int)b * (int)magic ^ (int)(b >> 32) * (int)(magic >> 32)) >> (32 - bits);
152#else
153 return static_cast<int>((b * magic) >> (64 - bits));
154#endif
155}

157uint64_t find_magic(int sq, int m, int bishop) {
158 uint64_t mask, b[4096], a[4096], used[4096], magic;
159 int i, j, k, n, fail;
160
161 mask = bishop ? bmask(sq) : rmask(sq);
162 n = count_1s(mask);
163
164 for (i = 0; i < (1 << n); i++) {
165 b[i] = index_to_uint64_t(i, n, mask);
166 a[i] = bishop ? batt(sq, b[i]) : ratt(sq, b[i]);
167 }
168 for (k = 0; k < 100000000; k++) {
169 magic = random_uint64_t_fewbits();
170 if (count_1s((mask * magic) & 0xFF00000000000000ULL) < 6) continue;
171 for (i = 0; i < 4096; i++) used[i] = 0ULL;
172 for (i = 0, fail = 0; !fail && i < (1 << n); i++) {
173 j = transform(b[i], magic, m);
174 if (used[j] == 0ULL)
175 used[j] = a[i];
176 else if (used[j] != a[i])
177 fail = 1;
178 }
179 if (!fail) return magic;
180 }
181 printf("***Failed***\n");
182 return 0ULL;
183}

185int count_bits(uint64_t bitboard) {
186 // bit size
187 int count = 0;
188
189 // pop bits untill bitboard is empty
190 while (bitboard) {
191 // increment size
192 count++;
193
194 // consecutively reset least significant 1st bit
195 bitboard &= bitboard - 1;
196 }
197
198 // return bit size
199 return count;
200}

203int get_ls1b_index(uint64_t bitboard) {
204 // make sure bitboard is not empty
205 if (bitboard != 0)
206 // convert trailing zeros before LS1B to ones and size them
207 return count_bits((bitboard & -bitboard) - 1);
208
209 // otherwise
210 // return illegal index
211 return -1;
212}

214uint64_t set_occupancy(int index, int bits_in_mask, uint64_t attack_mask) {
215 // occupancy map
216 uint64_t occupancy = 0ULL;
217
218 // loop over the range of bits within attack mask
219 for (int count = 0; count < bits_in_mask; count++) {
220 // get LS1B index of attacks mask
221 int8_t square = get_ls1b_index(attack_mask);
222
223 // pop LS1B in attack map
224 pop_bit(attack_mask, square);
225
226 // make sure occupancy is on board
227 if (index & (1 << count))
228 // populate occupancy map
229 occupancy |= (1ULL << square);
230 }
231
232 // return occupancy map
233 return occupancy;
234}

352void init_sliders_attacks(int is_bishop) {
353 // loop over 64 board squares
354 for (int8_t square = 0; square < 64; square++) {
355 // init bishop & rook masks
356 bishop_masks[square] = mask_bishop_attacks(square);
357 rook_masks[square] = mask_rook_attacks(square);
358
359 // init current mask
360 uint64_t mask = is_bishop ? mask_bishop_attacks(square) : mask_rook_attacks(square);
361
362 // size attack mask bits
363 int bit_count = count_bits(mask);
364
365 // occupancy variations size
366 int occupancy_variations = 1 << bit_count;
367
368 // loop over occupancy variations
369 for (int count = 0; count < occupancy_variations; count++) {
370 // bishop
371 if (is_bishop) {
372 // init occupancies, magic index & attacks
373 uint64_t occupancy = set_occupancy(count, bit_count, mask);
374 uint64_t magic_index = occupancy * bishopMagics[square] >> (64 - BBits[square]);
375 bishop_attacks[square][magic_index] = bishop_attacks_on_the_fly(square, occupancy);
376 }
377
378 // rook
379 else {
380 // init occupancies, magic index & attacks
381 uint64_t occupancy = set_occupancy(count, bit_count, mask);
382 uint64_t magic_index = occupancy * rookMagics[square] >> (64 - RBits[square]);
383 rook_attacks[square][magic_index] = rook_attacks_on_the_fly(square, occupancy);
384 }
385 }
386 }

416void findFastestSeed() {
417 // variable with max double
418 uint64_t max = std::numeric_limits<uint64_t>::max();
419 std::random_device seedRd;
420 std::mt19937_64 seedGen(seedRd());
421 std::uniform_int_distribution<uint64_t> seedDis;
422
423 while (true) {
424 uint64_t seed = seedDis(seedGen);
425 uint64_t total = 0;
426
427 for (int i = 0; i < 50; i++) {
428 gen.seed(seedDis(seedGen));
429 std::chrono::time_point<std::chrono::steady_clock> start =
430 std::chrono::steady_clock::now();
431
432 generateMagics();
433
434 std::chrono::time_point<std::chrono::steady_clock> now =
435 std::chrono::steady_clock::now();
436 uint64_t elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(now - start).
437 count();
438 total += elapsed;
439 }
440
441 uint64_t average = total / 50;
442 if (average < max) {
443 max = average;
444 std::cout << "New fastest seed: 0x" << std::uppercase << std::hex << seed <<
445 std::nouppercase
446 << std::dec << "ULL, took: " << average << "ms (average)" << std::endl;
447 }
