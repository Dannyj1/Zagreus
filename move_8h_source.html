66// bits 0-5: from square (0-63)
67// bits 6-11: to square (0-63)
68// bits 12-13: move type (00 = normal, 01 = promotion, 10 = en passant, 11 = castling)
69// bits 14-15: promotion piece (00 = queen, 01 = rook, 10 = bishop, 11 = knight)
70inline Move encodeMove(const uint8_t fromSquare, const uint8_t toSquare) {
71 // Assume normal move, so bits 12-13 are 00 and bits 14-15 are 00
72 return static_cast<Move>(fromSquare | (toSquare << 6));
73}
74
75inline Move encodeMove(const uint8_t fromSquare, const uint8_t toSquare, const MoveType moveType) {
76 return static_cast<Move>(fromSquare | (toSquare << 6) | (moveType << 12));
77}

79inline Move encodeMove(const uint8_t fromSquare, const uint8_t toSquare, const PromotionPiece promotionPiece) {
80 return static_cast<Move>(fromSquare | (toSquare << 6) | (0b01 << 12) | (promotionPiece << 14));
81}
