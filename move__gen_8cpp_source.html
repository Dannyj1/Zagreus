31template <PieceColor color, GenerationType type>
32void generateMoves(const Board& board, MoveList& moves) {
33 assert(moves.size == 0);
34
35 // TODO: Implement GenerationType logic using a mask that is computed based on type
36 constexpr Piece opponentKing = color == WHITE ? BLACK_KING : WHITE_KING;
37 const uint64_t ownPieces = board.getColorBitboard<color>();
38 const uint64_t opponentKingBB = board.getBitboard<opponentKing>();
39 const uint64_t genMask = ~(ownPieces | opponentKingBB);
40
41 generatePawnMoves<color, type>(board, moves, genMask);
42 generateKnightMoves<color, type>(board, moves, genMask);
43 generateBishopMoves<color, type>(board, moves, genMask);
44 generateRookMoves<color, type>(board, moves, genMask);
45 generateQueenMoves<color, type>(board, moves, genMask);
46 generateKingMoves<color, type>(board, moves, genMask);
47}

49template <PieceColor color, GenerationType type>
50void generatePawnMoves(const Board& board, MoveList& moves, const uint64_t genMask) {
51 constexpr Piece pawn = color == WHITE ? WHITE_PAWN : BLACK_PAWN;
52 constexpr PieceColor opponentColor = !color;
53
54 const uint64_t pawnBB = board.getBitboard<pawn>();
55 const uint64_t emptyBB = board.getEmptyBitboard();
56 const uint64_t opponentPieces = board.getColorBitboard<opponentColor>();
57 uint64_t pawnSinglePushes;
58 uint64_t pawnDoublePushes;
59 uint64_t pawnWestAttacks;
60 uint64_t pawnEastAttacks;
61
62 if constexpr (color == WHITE) {
63 pawnSinglePushes = whitePawnSinglePush(pawnBB, emptyBB);
64 pawnDoublePushes = whitePawnDoublePush(pawnBB, emptyBB);
65 pawnWestAttacks = whitePawnWestAttacks(pawnBB);
66 pawnEastAttacks = whitePawnEastAttacks(pawnBB);
67 } else {
68 pawnSinglePushes = blackPawnSinglePush(pawnBB, emptyBB);
69 pawnDoublePushes = blackPawnDoublePush(pawnBB, emptyBB);
70 pawnWestAttacks = blackPawnWestAttacks(pawnBB);
71 pawnEastAttacks = blackPawnEastAttacks(pawnBB);
72 }
73
74 uint64_t enPassantMask = squareToBitboard(board.getEnPassantSquare());
75
76 if constexpr (color == WHITE) {
77 enPassantMask &= RANK_6;
78 } else {
79 enPassantMask &= RANK_3;
80 }
81
82 pawnSinglePushes &= genMask;
83 pawnDoublePushes &= genMask;
84 pawnWestAttacks &= (opponentPieces | enPassantMask) & genMask;
85 pawnEastAttacks &= (opponentPieces | enPassantMask) & genMask;
86
87 constexpr Direction fromPushDirection = color == WHITE ? NORTH : SOUTH;
88 constexpr Direction fromSqWestAttackDirection = color == WHITE ? NORTH_WEST : SOUTH_WEST;
89 constexpr Direction fromSqEastAttackDirection = color == WHITE ? NORTH_EAST : SOUTH_EAST;
90 constexpr uint64_t promotionRank = color == WHITE ? RANK_8 : RANK_1;
91
92 while (pawnSinglePushes) {
93 const uint8_t squareTo = popLsb(pawnSinglePushes);
94 const uint64_t squareToBB = squareToBitboard(squareTo);
95 const uint8_t squareFrom = squareTo - fromPushDirection;
96
97 if (squareToBB & promotionRank) {
98 for (const PromotionPiece promotionPiece : {QUEEN_PROMOTION, ROOK_PROMOTION, BISHOP_PROMOTION,
99 KNIGHT_PROMOTION}) {
100 const Move move = encodeMove(squareFrom, squareTo, promotionPiece);
101 moves.moves[moves.size] = move;
102 moves.size++;
103 }
104 } else {
105 const Move move = encodeMove(squareFrom, squareTo);
106
107 moves.moves[moves.size] = move;
108 moves.size++;
109 }
110 }
111
112 while (pawnDoublePushes) {
113 const uint8_t squareTo = popLsb(pawnDoublePushes);
114 const uint8_t squareFrom = squareTo - fromPushDirection - fromPushDirection;
115 const Move move = encodeMove(squareFrom, squareTo);
116
117 moves.moves[moves.size] = move;
118 moves.size++;
119 }
120
121 while (pawnWestAttacks) {
122 const uint8_t squareTo = popLsb(pawnWestAttacks);
123 const uint64_t squareToBB = squareToBitboard(squareTo);
124 const uint8_t squareFrom = squareTo - fromSqWestAttackDirection;
125
126 if (squareTo == board.getEnPassantSquare() && squareToBB & enPassantMask) {
127 const Move move = encodeMove(squareFrom, squareTo, EN_PASSANT);
128 moves.moves[moves.size] = move;
129 moves.size++;
130 } else {
131 if (squareToBB & promotionRank) {
132 for (const PromotionPiece promotionPiece : {QUEEN_PROMOTION, ROOK_PROMOTION, BISHOP_PROMOTION,
133 KNIGHT_PROMOTION}) {
134 const Move move = encodeMove(squareFrom, squareTo, promotionPiece);
135 moves.moves[moves.size] = move;
136 moves.size++;
137 }
138 } else {
139 const Move move = encodeMove(squareFrom, squareTo);
140 moves.moves[moves.size] = move;
141 moves.size++;
142 }
143 }
144 }
145
146 while (pawnEastAttacks) {
147 const uint8_t squareTo = popLsb(pawnEastAttacks);
148 const uint64_t squareToBB = squareToBitboard(squareTo);
149 const uint8_t squareFrom = squareTo - fromSqEastAttackDirection;
150
151 if (squareTo == board.getEnPassantSquare() && squareToBB & enPassantMask) {
152 const Move move = encodeMove(squareFrom, squareTo, EN_PASSANT);
153 moves.moves[moves.size] = move;
154 moves.size++;
155 } else {
156 if (squareToBB & promotionRank) {
157 for (const PromotionPiece promotionPiece : {QUEEN_PROMOTION, ROOK_PROMOTION, BISHOP_PROMOTION,
158 KNIGHT_PROMOTION}) {
159 const Move move = encodeMove(squareFrom, squareTo, promotionPiece);
160 moves.moves[moves.size] = move;
161 moves.size++;
162 }
163 } else {
164 const Move move = encodeMove(squareFrom, squareTo);
165 moves.moves[moves.size] = move;
166 moves.size++;
167 }
168 }
169 }
170}

172template <PieceColor color, GenerationType type>
173void generateKnightMoves(const Board& board, MoveList& moves, const uint64_t genMask) {
174 constexpr Piece knight = color == WHITE ? WHITE_KNIGHT : BLACK_KNIGHT;
175 uint64_t knightBB = board.getBitboard<knight>();
176
177 while (knightBB) {
178 const uint8_t fromSquare = popLsb(knightBB);
179 const uint64_t attacks = getKnightAttacks(fromSquare);
180 uint64_t genBB = attacks & genMask;
181
182 while (genBB) {
183 const uint8_t toSquare = popLsb(genBB);
184 const Move move = encodeMove(fromSquare, toSquare);
185
186 moves.moves[moves.size] = move;
187 moves.size++;
188 }
189 }
190}

252template <PieceColor color, GenerationType type>
253void generateKingMoves(const Board& board, MoveList& moves, const uint64_t genMask) {
254 constexpr Piece king = color == WHITE ? WHITE_KING : BLACK_KING;
255 uint64_t kingBB = board.getBitboard<king>();
256 const uint8_t fromSquare = popLsb(kingBB);
257 uint64_t genBB = getKingAttacks(fromSquare) & genMask;
258
259 while (genBB) {
260 const uint8_t toSquare = popLsb(genBB);
261 const Move move = encodeMove(fromSquare, toSquare);
262
263 moves.moves[moves.size] = move;
264 moves.size++;
265 }
266
267 const uint8_t castlingRights = board.getCastlingRights();
268
269 if constexpr (color == WHITE) {
270 if (castlingRights & WHITE_KINGSIDE && board.canCastle<WHITE_KINGSIDE>()) {
271 const Move move = encodeMove(E1, G1, CASTLING);
272
273 moves.moves[moves.size] = move;
274 moves.size++;
275 }
276
277 if (castlingRights & WHITE_QUEENSIDE && board.canCastle<WHITE_QUEENSIDE>()) {
278 const Move move = encodeMove(E1, C1, CASTLING);
279
280 moves.moves[moves.size] = move;
281 moves.size++;
282 }
283 } else if constexpr (color == BLACK) {
284 if (castlingRights & BLACK_KINGSIDE && board.canCastle<BLACK_KINGSIDE>()) {
285 const Move move = encodeMove(E8, G8, CASTLING);
286
287 moves.moves[moves.size] = move;
288 moves.size++;
289 }
290
291 if (castlingRights & BLACK_QUEENSIDE && board.canCastle<BLACK_QUEENSIDE>()) {
292 const Move move = encodeMove(E8, C8, CASTLING);
293
294 moves.moves[moves.size] = move;
295 moves.size++;
296 }
297 }
