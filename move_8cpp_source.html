54std::string getMoveNotation(const Move move) {
55 const uint8_t fromSquare = getFromSquare(move);
56 const uint8_t toSquare = getToSquare(move);
57
58 if (const MoveType moveType = getMoveType(move);
59 moveType == PROMOTION) {
60 const PromotionPiece promotionPiece = getPromotionPiece(move);
61
62 return getMoveNotation(fromSquare, toSquare, promotionPiece);
63 }
64
65 return getMoveNotation(fromSquare, toSquare);
66}

68Move fromMoveNotation(const std::string_view notation) {
69 const Square fromSquare = fromSquareNotation(notation.substr(0, 2));
70 const Square toSquare = fromSquareNotation(notation.substr(2, 2));
71
72 // If there is a 5th character, it is a promotion
73 if (notation.size() == 5) {
74 const char promotionPieceChar = notation[4];
75 PromotionPiece promotionPiece;
76
77 switch (promotionPieceChar) {
78 case 'q':
79 promotionPiece = QUEEN_PROMOTION;
80 break;
81 case 'r':
82 promotionPiece = ROOK_PROMOTION;
83 break;
84 case 'b':
85 promotionPiece = BISHOP_PROMOTION;
86 break;
87 case 'n':
88 promotionPiece = KNIGHT_PROMOTION;
89 break;
90 default:
91 assert(false);
92 return 0;
93 }
94
95 return encodeMove(fromSquare, toSquare, promotionPiece);
96 }
97
98 return encodeMove(fromSquare, toSquare);
99}
